FUNC void printEffects(Effects e) {
        int i;
        FOR i = 0 TO e.getCount() STEP 2 {
                Effect ef;
                ef = e.getAt(i);
                int j;
                j = ef.getParamValue("value");
                PRINT("value: " + j);
                e.delete(i);
        }
}

FUNC int power(int i) {
    int sum;
    sum = 1;
    IF (i != 0) {
        int tmp;
        int tmpi;
        tmpi = i - 1;
        tmp = power(tmpi);
        sum = tmp + sum;
        RETURN sum;
    }
    RETURN sum;
}

    Song song;

    int i;

    int res;
    res = power(2);
    PRINT(res);

    IF (i > 0)

}
