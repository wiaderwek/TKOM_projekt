FUNC void modifySeparately(Effects e, Song s) {
        int i;
        PRINT(song.getLength());
        FOR i = 0 TO e.getCount() STEP 2 {
                Effect ef = e.getAt(i);
                Effects efs;
                efs.add(ef);
                song = song.modify(efs);
                PRINT(song.getLength());
        }
}

Song song;
song.load("Mama mia");
Effects e;
Effect ex;
ex.setType("Extend");
ex.setParam("value", 4);
Effect rise;
rise.setType("Rise");
rise.setParam("value", 4);

e.add(ex);
e.add(rise);

modifySeparately(e, song);
